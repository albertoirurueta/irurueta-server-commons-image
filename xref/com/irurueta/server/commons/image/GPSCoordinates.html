16  package com.irurueta.server.commons.image;
17  
18  /**
19   * This class contains GPS coordinates of an image (latitude, longitude and
20   * altitude) expressed in degrees and meters respectively.
21   */
22  public class GPSCoordinates {
23      
24      /**
25       * Latitude of location expressed in degrees.
26       * Note: minutes and seconds must be expressed as a decimal fraction of a 
27       * degree.
28       */    
29      private Double mLatitude;
30      
31      /**
32       * Longitude of location expressed in degrees.
33       * Note: minutes and seconds must be expressed a decimal fraction of a 
34       * degree.
35       */
36      private Double mLongitude;
37      
38      /**
39       * Altitude of location expressed in meters.
40       * Zero indicates sea level. Positive values indicate an altitude above
41       * sea level and negative values indicate an altitude below sea level.
42       */
43      private Double mAltitude;
44      
45      /**
46       * Constructor.
47       */
48      protected GPSCoordinates() { }
49      
50      /**
51       * Constructor.
52       * @param latitude latitude expressed in degrees. Note: minutes and seconds
53       * must be expressed as a decimal fraction of a degree.
54       * @param longitude longitude expressed in degrees. Note: minutes and 
55       * seconds must be expressed as a decimal fraction of a degree.
56       */
57      public GPSCoordinates(double latitude, double longitude) {
58          mLatitude = latitude;
59          mLongitude = longitude;
60      }
61      
62      /**
63       * Constructor.
64       * @param latitude latitude expressed in degrees. Note: minutes and seconds
65       * must be expressed as a decimal fraction of a degree.
66       * @param longitude longitude expressed in degrees. Note: minutes and 
67       * seconds must be expressed as a decimal fraction of a degree.
68       * @param altitude altitude expressed in meters.
69       */
70      public GPSCoordinates(double latitude, double longitude, 
71              double altitude) {
72          mLatitude = latitude;
73          mLongitude = longitude;
74          mAltitude = altitude;
75      }
76          
77      /**
78       * Returns latitude expressed in degrees, or null if value is unknown. 
79       * Note: minutes and seconds must be expressed as a decimal fraction of a 
80       * degree.
81       * @return latitude expressed in degrees or null if unknown.
82       */
83      public Double getLatitude() {
84          return mLatitude;
85      }
86      
87      /**
88       * Sets latitude expressed in degrees. Null indicates that value is unknown
89       * Note: minutes and seconds must be expressed as a decimal fraction of a 
90       * degree.
91       * @param latitude latitude to be set expressed in degrees or null if 
92       * unknown.
93       */
94      public void setLatitude(double latitude) {
95          mLatitude = latitude;
96      }
97      
98      /**
99       * Indicates whether latitude coordinate is available.
100      * @return true if available, false otherwise.
101      */
102     public boolean isLatitudeAvailable() {
103         return mLatitude != null;
104     }
105             
106     /**
107      * Returns longitude expressed in degrees, or null if value is unknown. 
108      * Note: minutes and seconds must be expressed as a decimal
109      * @return longitude expressed in degrees or null if unknown.
110      */
111     public Double getLongitude() {
112         return mLongitude;
113     }
114     
115     /**
116      * Sets longitude expressed in degrees. Null indicates that value is 
117      * unknown.
118      * Note: minutes and seconds must be expressed as a decimal faction of a 
119      * degree.
120      * @param longitude longitude to be set expressed in degrees or null if 
121      * unknown.
122      */
123     public void setLongitude(double longitude) {
124         mLongitude = longitude;
125     }
126     
127     /**
128      * Indicates whether longitude coordinate is available.
129      * @return true if available, false otherwise.
130      */
131     public boolean isLongitudeAvailable() {
132         return mLongitude != null;
133     }
134             
135     /**
136      * Returns altitude expressed in meters, or null if value is unknown.
137      * Zero indicates sea level. Positive values indicate an altitude above
138      * sea level and negative values indicate an altitude below sea level.
139      * @return altitude expressed in meters or null if unknown.
140      */
141     public Double getAltitude() {
142         return mAltitude;
143     }
144     
145     /**
146      * Sets altitude expressed in meters. Null indicates that value is unknown.
147      * @param altitude altitude expressed in meters or null if unknown.
148      */
149     public void setAltitude(Double altitude) {
150         mAltitude = altitude;
151     }        
152     
153     /**
154      * Indicates whether longitude coordinate is available.
155      * @return true if available, false otherwise.
156      */
157     public boolean isAltitudeAvailable() {
158         return mAltitude != null;
159     }
160 }
