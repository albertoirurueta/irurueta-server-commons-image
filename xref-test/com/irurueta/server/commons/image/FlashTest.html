16  package com.irurueta.server.commons.image;
17  
18  import org.junit.Test;
19  
20  import static org.junit.Assert.assertEquals;
21  
22  public class FlashTest {
23  
24      @Test
25      public void testFromValue() {
26          assertEquals(Flash.fromValue(0x0000), Flash.FLASH_DID_NOT_FIRE);
27          assertEquals(Flash.fromValue(0x0001), Flash.FLASH_FIRED);
28          assertEquals(Flash.fromValue(0x0005),
29                  Flash.STROBE_RETURN_LIGHT_NOT_DETECTED);
30          assertEquals(Flash.fromValue(0x0007),
31                  Flash.STROBE_RETURN_LIGHT_DETECTED);
32          assertEquals(Flash.fromValue(0x0009),
33                  Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE);
34          assertEquals(Flash.fromValue(0x000D),
35                  Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE_RETURN_LIGHT_NOT_DETECTED);
36          assertEquals(Flash.fromValue(0x000F),
37                  Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE_RETURN_LIGHT_DETECTED);
38          assertEquals(Flash.fromValue(0x0010),
39                  Flash.FLASH_DID_NOT_FIRE_COMPULSORY_FLASH_MODE);
40          assertEquals(Flash.fromValue(0x0018),
41                  Flash.FLASH_DID_NOT_FIRE_AUTO_MODE);
42          assertEquals(Flash.fromValue(0x0019),
43                  Flash.FLASH_FIRED_AUTO_MODE);
44          assertEquals(Flash.fromValue(0x001D),
45                  Flash.FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_NOT_DETECTED);
46          assertEquals(Flash.fromValue(0x001F),
47                  Flash.FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_DETECTED);
48          assertEquals(Flash.fromValue(0x0020), Flash.NO_FLASH_FUNCTION);
49          assertEquals(Flash.fromValue(0x0041),
50                  Flash.FLASH_FIRED_RED_EYE_REDUCTION_MODE);
51          assertEquals(Flash.fromValue(0x0045),
52                  Flash.FLASH_FIRED_RED_EYE_REDUCTION_MODE_RETURN_LIGHT_NOT_DETECTED);
53          assertEquals(Flash.fromValue(0x0047),
54                  Flash.FLASH_FIRED_RED_EYE_REDUCTION_MODE_RETURN_LIGHT_DETECTED);
55          assertEquals(Flash.fromValue(0x0049),
56                  Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE_RED_EYE_REDUCTION_MODE);
57          assertEquals(Flash.fromValue(0x004D),
58                  Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE_RED_EYE_REDUCTION_MODE_RETURN_LIGHT_NOT_DETECTED);
59          assertEquals(Flash.fromValue(0x004F),
60                  Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE_RED_EYE_REDUCTION_MODE_RETURN_LIGHT_DETECTED);
61          assertEquals(Flash.fromValue(0x0059),
62                  Flash.FLASH_FIRED_AUTO_MODE_RED_EYE_REDUCTION_MODE);
63          assertEquals(Flash.fromValue(0x005D),
64                  Flash.FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_NOT_DETECTED_RED_EYE_REDUCTION_MODE);
65          assertEquals(Flash.fromValue(0x005F),
66                  Flash.FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_DETECTED_RED_EYE_REDUCTION_MODE);
67          assertEquals(Flash.fromValue(0xFFFF), Flash.UNKNOWN);
68      }
69  
70      @Test
71      public void testGetValue() {
72          assertEquals(Flash.FLASH_DID_NOT_FIRE.getValue(), 0x0000);
73          assertEquals(Flash.FLASH_FIRED.getValue(), 0x0001);
74          assertEquals(Flash.STROBE_RETURN_LIGHT_NOT_DETECTED.getValue(), 0x0005);
75          assertEquals(Flash.STROBE_RETURN_LIGHT_DETECTED.getValue(), 0x0007);
76          assertEquals(Flash.FLASH_FIRED_COMPULSORY_FLASH_MODE.getValue(),
77                  0x0009);
78          assertEquals(Flash.
79                  FLASH_FIRED_COMPULSORY_FLASH_MODE_RETURN_LIGHT_NOT_DETECTED.
80                  getValue(), 0x000D);
81          assertEquals(Flash.
82                  FLASH_FIRED_COMPULSORY_FLASH_MODE_RETURN_LIGHT_DETECTED.

85                  0x0010);

87          assertEquals(Flash.FLASH_FIRED_AUTO_MODE.getValue(), 0x0019);
88          assertEquals(Flash.FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_NOT_DETECTED.
89                  getValue(), 0x001D);
90          assertEquals(Flash.FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_DETECTED.
91                  getValue(), 0x001F);
92          assertEquals(Flash.NO_FLASH_FUNCTION.getValue(), 0x0020);
93          assertEquals(Flash.FLASH_FIRED_RED_EYE_REDUCTION_MODE.getValue(),

97                  getValue(), 0x0045);

111                 getValue(), 0x0059);
112         assertEquals(Flash.
113                 FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_NOT_DETECTED_RED_EYE_REDUCTION_MODE.
114                 getValue(), 0x005D);
115         assertEquals(Flash.
116                 FLASH_FIRED_AUTO_MODE_RETURN_LIGHT_DETECTED_RED_EYE_REDUCTION_MODE.
117                 getValue(), 0x005F);
